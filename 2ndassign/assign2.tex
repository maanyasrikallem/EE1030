\textbf{Question}:\\
The coordinates of the three consecutive vertices of a parallelogram $ABCD$ are $A$ (1,3),$B$(-1,2), and $C$(2,5).Find the coordinates of the fourth vertex $D$.
\hfill(10,2021)
\\ \textbf{Sol:}

 \begin{align}
	 A-B=D-C

		 x-2
		 \\
		 y-5
	 }
	 \\ x=4 , y=6
	 \\ D = \myvec{
		 4

		 6
	 }
 \end{align}
